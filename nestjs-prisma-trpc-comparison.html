@Injectable()

  async createUserWithProfile(userData: UserCreateDto, profileData: ProfileCreateDto) {
    return this.prisma.$transaction(async (tx) => {
      const user = await tx.user.create({
        data: userData,
      });

      const profile = await tx.profile.create({
        data: {

          userId: user.id,
        },
      });

      return { user, profile };
    });

}
